import io ;
import FacInterface;

class Fac extends FacInterface{
    public int computeFac ( int num , boolean stop){
        int num_aux ;
        if ( !(num < 1))
            num_aux = 1;
        else
            num_aux = num * (this.computeFac(num-1));
        return num_aux ;
    }

    public static void main ( String [] args ){
        io.println( new Fac().computeFac(10)); /* assuming the existence?*/

}
}
